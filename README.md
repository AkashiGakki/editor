## Usage

### In vue

```shell
pnpm i @use-kit/editor-vue
```

```vue
<script setup lang="ts">
import TextEditor from '@use-kit/editor-vue'
</script>

<template>
  <TextEditor />
</template>
```

## In react

```shell
pnpm i @use-kit/editor-react
```

```tsx
import TextEditor from '@use-kit/editor-react'

const EditorView = () => {
  return (
    <>
      <TextEditor />
    </>
  )
}

export default EditorView
```
